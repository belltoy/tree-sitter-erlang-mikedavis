================================================================================
Empty default sigil
================================================================================

~""

--------------------------------------------------------------------------------

(source
  (sigil))

================================================================================
Lowercase sigil allows escapes
================================================================================

~b"Escapes\n"

--------------------------------------------------------------------------------

(source
  (sigil
    (sigil_prefix)
    (quoted_content)
    (escape_sequence)))

================================================================================
Uppercase sigil reads verbatim
================================================================================

~B"Escapes\n"

--------------------------------------------------------------------------------

(source
  (sigil
    (sigil_prefix)
    (quoted_content)))

================================================================================
Triple quoted sigil
================================================================================

~B"""
This is a triple quoted string binary.
All contents are verbatim, even this: ~r"abc\d+"i.
"""

--------------------------------------------------------------------------------

(source
  (sigil
    (sigil_prefix)
    (quoted_content)))
